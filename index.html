float mouseXnorm = 0.5; ArrayList particles = new ArrayList(); int particleCount = 5000; float auroraOffset = 0; PFont chineseFont; void setup() { size(900, 600); colorMode(RGB, 255); smooth(); chineseFont = createFont("Microsoft YaHei", 68); // Mac 用下面这行 // chineseFont = createFont("PingFang SC", 68); for (int i = 0; i < particleCount; i++) { particles.add(new Particle()); } } void draw() { drawBlackGradient(); drawAurora(); updateParticles(); drawTypography(); timeOffset += 0.8; mouseXnorm = constrain(mouseX / (float)width, 0, 1); } void drawBlackGradient() { for (int i = 0; i <= height; i++) { float inter = map(i, 0, height, 0, 1); float brightness = 5 + 15 * sin(inter * PI); stroke(brightness); line(0, i, width, i); } } // ========== 单道统一方向极光 ========== void drawAurora() { pushMatrix(); translate(0, height * 0.25); // 只保留一组主极光 for (int i = 0; i < 10; i++) { float alphaVal = 30 - i * 2.5; float wave = sin(radians(timeOffset * 1.2 + i * 15)) * 18; stroke(190 + i * 6, max(8, alphaVal)); strokeWeight(2.2 - i * 0.15); noFill(); beginShape(); for (float x = 0; x <= width; x += 10) { float y = 60 + sin(radians(x * 0.32 + timeOffset * 1.5)) * 22 + wave * sin(radians(x * 0.08)); y += (mouseXnorm - 0.5) * 35 * sin(radians(x * 0.12)); vertex(x, y + i * 12); } endShape(); } popMatrix(); } class Particle { float x, y, vx, vy, size, alpha; Particle() { reset(); } void reset() { x = random(width); y = random(height); vx = random(-0.4, 0.4); vy = random(-0.3, -0.9); size = random(1.0, 2.8); alpha = random(50, 140); } void update() { x += vx; y += vy; if (x < -20 || x > width + 20 || y < -20 || y > height + 20) { reset(); x = random(width); y = height + 10; } float dx = x - mouseX; float dy = y - mouseY; float dist = sqrt(dx*dx + dy*dy); if (dist < 60 && dist > 0) { vx += dx / dist * 0.1; vy += dy / dist * 0.1; } vx *= 0.99; vy *= 0.99; alpha = 60 + sin(radians(frameCount * 2 + x)) * 35; } void draw() { stroke(255, alpha); strokeWeight(size); point(x, y); } } void updateParticles() { for (Particle p : particles) { p.update(); p.draw(); } } void drawTypography() { resetMatrix(); float titleShiftY = sin(radians(timeOffset * 1.5)) * 2; textAlign(CENTER, CENTER); textFont(chineseFont); textSize(68); fill(255); textLeading(80); text("创意编程\n作业展", width/2, height * 0.38 + titleShiftY); textSize(15); fill(245, 245, 255, 235); text("CREATIVE CODING EXHIBITION", width/2, height * 0.54); float lineWidth = 220 + sin(radians(timeOffset * 2.8)) * 20; stroke(255, 230); strokeWeight(1.5); line(width/2 - lineWidth/2, height * 0.585, width/2 + lineWidth/2, height * 0.585); textSize(16); fill(240, 240, 255, 240); textLeading(28); text("2026.04.20 - 04.26\n长沙 · 梅溪湖艺术中心", width/2, height * 0.72); textSize(11); fill(200, 200, 220, 190); text("交互设计工作室 ｜ 新媒体艺术系", width/2, height - 48); }
